Circuit Analysis

This circuit is a robust synchronizer, noise filter (debouncer), and negative-edge detector. It is specifically designed to handle a slow, potentially noisy external signal (like a PS/2 clock) and safely interface it with a much faster clock domain (e.g.,50MHz).

The design consists of three main stages: 1. Asynchronous Input Synchronization Utilizing 2-DFF Synchronizer: The first two D-flip-flops synchronize the asynchronous PS2C input to the CLK50MHZ clock domain, preventing metastability issues. 2. Noise Filter / Debouncer: An 8-bit shift register continually samples the synchronized signal. An 8-input AND gate checks if all 8 bits are '1' (stable HIGH). An 8-input NOR gate checks if all 8 bits are '0' (stable LOW). A 4-to-1 Multiplexer controls the next state (SC): If all bits are '1' ($S_1=1, S_0=0 \rightarrow \text{Select } I2='1'$), SC becomes 1. If all bits are '0' ($S_1=0, S_0=1 \rightarrow \text{Select } I1='0'$), SC becomes 0. If the bits are mixed '0' and '1' ($S_1=0, S_0=0 \rightarrow \text{Select } I0$), SC holds its previous value (DSC). This ignores any glitches or noise shorter than 8 clock cycles. 3. Negative (Falling) Edge Detection: The final D-flip-flop stores the previous filtered state (DSC). The output ne goes high for exactly one clock cycle when the current state SC is low (~SC) and the previous state DSC is high.

SystemVerilog Design

`timescale 1ns / 1ps

module ps2_noise_filter (
    input  logic CLK50MHZ,
    input  logic resetn,
    input  logic PS2C,
    output logic ne
);

    // 1. Synchronization FFs
    logic ps2c_sync1, ps2c_sync2;

    always_ff @(posedge CLK50MHZ or negedge resetn) begin
        if (!resetn) begin
            ps2c_sync1 <= 1'b1; // Idle state for PS/2 clock is high
            ps2c_sync2 <= 1'b1;
        end else begin
            ps2c_sync1 <= PS2C;
            ps2c_sync2 <= ps2c_sync1;
        end
    end

    // 2. Noise Filter (8-bit Shift Register)
    logic [7:0] q;

    always_ff @(posedge CLK50MHZ or negedge resetn) begin
        if (!resetn) begin
            q <= 8'hFF; // Initialize to all 1s (Idle state)
        end else begin
            q <= {q[6:0], ps2c_sync2}; // Shift left, inserting newest sample
        end
    end

    // Reduction logic for MUX select lines
    logic s1, s0;
    assign s1 = &q;  // 8-input AND: 1 if all bits are 1
    assign s0 = ~|q; // 8-input NOR: 1 if all bits are 0

    // Multiplexer logic
    logic sc, dsc;

    always_comb begin
        case ({s1, s0})
            2'b00: sc = dsc;  // I0: Mixed bits -> hold previous state
            2'b01: sc = 1'b0; // I1: All 0s     -> stable low
            2'b10: sc = 1'b1; // I2: All 1s     -> stable high
            2'b11: sc = dsc;  // I3: Impossible -> hold state (fallback)
        endcase
    end

    // 3. Edge Detection FF and Logic
    always_ff @(posedge CLK50MHZ or negedge resetn) begin
        if (!resetn) begin
            dsc <= 1'b1;
        end else begin
            dsc <= sc;
        end
    end

    // Negative edge detected when SC transitions to 0 while DSC is 1
    assign ne = (~sc) & dsc;

endmodule

SystemVerilog Self-Checking Testbench

This testbench models the 50MHz clock, injects high-frequency noise spikes (faults) onto the PS2C line that the filter is designed to reject, and then sends valid 256KHz negative edges to verify detection. It checks if the exact number of expected negative edges occurs.

`timescale 1ns / 1ps

module tb_ps2_noise_filter();

    // Testbench signals
    logic CLK50MHZ;
    logic resetn;
    logic PS2C;
    logic ne;

    // Instantiate the Device Under Test (DUT)
    ps2_noise_filter dut (.*);

    // Clock Generation (50 MHz = 20ns period)
    initial begin
        CLK50MHZ = 0;
        forever #10 CLK50MHZ = ~CLK50MHZ; 
    end

    // Edge counter for self-checking
    int detected_edges = 0;
    always @(posedge CLK50MHZ) begin
        if (ne) detected_edges++;
    end

    // Stimulus and Fault Injection
    initial begin
        $display("Starting PS/2 Noise Filter Simulation...");

        // Initialize
        resetn = 0;
        PS2C = 1; // PS/2 lines IDLE HIGH
        #100;
        resetn = 1;
        #200; // Wait for system to stabilize

        // -------------------------------------------------------------
        // FAULT INJECTION: Glitches/Noise (Should NOT trigger 'ne')
        // -------------------------------------------------------------
        $display("[%0t] Injecting short noise glitches...", $time);

        // Fault 1: 3-cycle glitch (60ns) -> Filter needs 8 cycles, should ignore
        PS2C = 0; #60;  
        PS2C = 1; #500; 

        // Fault 2: 6-cycle glitch (120ns) -> Still < 8 cycles, should ignore
        PS2C = 0; #120; 
        PS2C = 1; #500;

        // -------------------------------------------------------------
        // VALID SIGNAL: Actual 256KHz Clock Edge
        // 256KHz period is ~3906ns. Low phase is ~1953ns.
        // -------------------------------------------------------------
        $display("[%0t] Simulating valid negative edge...", $time);

        // Valid negative edge (hold low for long enough to fill shift register)
        PS2C = 0; #1953; 

        // Return to high (simulating positive edge of 256KHz clock)
        PS2C = 1; #1953;

        // -------------------------------------------------------------
        // FAULT INJECTION 2: Glitch during a valid low phase
        // -------------------------------------------------------------
        $display("[%0t] Injecting noise during a valid cycle...", $time);
        PS2C = 0; #2000; // Valid negative edge

        // Spike high for 4 cycles in the middle of a low state (should be ignored)
        PS2C = 1; #80;   
        PS2C = 0; #1800; 

        PS2C = 1; #2000; // Return high

        // -------------------------------------------------------------
        // Self-Checking Verification
        // -------------------------------------------------------------
        $display("-------------------------------------------------");
        if (detected_edges == 2) begin
            $display("PASS: Circuit perfectly filtered faults and detected exactly 2 valid edges.");
        end else begin
            $error("FAIL: Expected 2 edges, but detected %0d edges.", detected_edges);
        end
        $display("-------------------------------------------------");

        $finish;
    end

    // Assertions to ensure 'ne' is strictly a one-cycle pulse
    property ne_is_one_cycle;
        @(posedge CLK50MHZ) ne |=> !ne;
    endproperty
    assert property (ne_is_one_cycle) else $error("Pulse 'ne' lasted longer than 1 clock cycle!");

endmodule

Step 2: Run Simulation Commands

Create a directory to store compiled design unit

ModelSim> vlib work

ModelSim> vlog -sv -work work ../src/ps2_noise_filter.sv
